Stored Procedures
CREATE PROCEDURE · INSERT · SELECT · RAISERROR
Los Stored Procedures encapsulan lógica de negocio reutilizable en el servidor. Aquí se muestran 5 SPs que cubren: validación de datos, INSERT condicional, y SELECT con JOIN + agregación.
SP INSERT SP_RegistrarDonacion — Donaciones.SP
▼ 
CREATE PROCEDURE Donaciones.SP_RegistrarDonacion
    @PersonaID  INT,
    @Monto      DECIMAL(12,2),
    @MetodoPago VARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;

    -- Validación: el monto debe ser positivo
    IF @Monto <= 0
    BEGIN
        RAISERROR('El monto debe ser mayor que 0.', 16, 1);
        RETURN;
    END

    -- DML: INSERT de la donación
    INSERT INTO Donaciones.Donacion
        (PersonaID, Monto, MetodoPago)
    VALUES
        (@PersonaID, @Monto, @MetodoPago);

    PRINT 'Donación registrada correctamente.';
END;

▼ 
CREATE PROCEDURE Personas.SP_RegistrarVoluntario
    @PersonaID   INT,
    @FechaIngreso DATE
AS
BEGIN
    SET NOCOUNT ON;

    -- Verifica duplicados antes de insertar
    IF EXISTS (SELECT 1 FROM Personas.Voluntario
               WHERE PersonaID = @PersonaID)
    BEGIN
        RAISERROR('La persona ya está registrada como voluntario.', 16, 1);
        RETURN;
    END

    -- DML: INSERT del voluntario
    INSERT INTO Personas.Voluntario
        (PersonaID, FechaIngreso)
    VALUES
        (@PersonaID, @FechaIngreso);

    PRINT 'Voluntario registrado correctamente.';
END;

CREATE PROCEDURE Voluntariado.SP_InscribirVoluntario
    @VoluntarioID INT,
    @ActividadID  INT
AS
BEGIN
    SET NOCOUNT ON;

    -- Subconsulta EXISTS para evitar duplicados
    IF EXISTS (
        SELECT 1
        FROM  Voluntariado.Inscripcion
        WHERE VoluntarioID = @VoluntarioID
          AND ActividadID  = @ActividadID
    )
    BEGIN
        RAISERROR('El voluntario ya está inscrito en esta actividad.', 16, 1);
        RETURN;
    END

    -- DML: INSERT de la inscripción
    INSERT INTO Voluntariado.Inscripcion
        (VoluntarioID, ActividadID)
    VALUES
        (@VoluntarioID, @ActividadID);

    PRINT 'Inscripción realizada correctamente.';
END;

SP SELECT SP_TotalDonadoPorPersona — Reporte con LEFT JOIN
▼ 
CREATE PROCEDURE Donaciones.SP_TotalDonadoPorPersona
    @PersonaID INT
AS
BEGIN
    SET NOCOUNT ON;

    -- SELECT con LEFT JOIN y agregación SUM
    SELECT
        p.PrimerNombre,
        p.PrimerApellido,
        ISNULL(SUM(d.Monto), 0) AS TotalDonado
    FROM       Personas.Persona   p
    LEFT JOIN  Donaciones.Donacion d
        ON p.PersonaID = d.PersonaID
    WHERE p.PersonaID = @PersonaID
    GROUP BY
        p.PrimerNombre,
        p.PrimerApellido;
END;

Funciones Escalares
CREATE FUNCTION · RETURNS · SELECT escalar
Las funciones escalares retornan un único valor calculado. Se usan dentro de SELECT, WHERE o como valores en otras consultas. Cubren: suma, promedio, conteo por entidad.
FN SELECT FN_TotalDonado · FN_PromedioDonaciones · FN_TotalInscritos · FN_TotalCertificados · FN_UsuariosActivos
▼ 
/* 1 — Total donado por una persona específica */
CREATE FUNCTION Donaciones.FN_TotalDonado(@PersonaID INT)
RETURNS DECIMAL(12,2)
AS
BEGIN
    DECLARE @Total DECIMAL(12,2);
    SELECT @Total = ISNULL(SUM(Monto), 0)
    FROM  Donaciones.Donacion
    WHERE PersonaID = @PersonaID;
    RETURN @Total;
END;
GO

/* 2 — Promedio global de todas las donaciones */
CREATE FUNCTION Donaciones.FN_PromedioDonaciones()
RETURNS DECIMAL(12,2)
AS
BEGIN
    DECLARE @Prom DECIMAL(12,2);
    SELECT @Prom = AVG(Monto) FROM Donaciones.Donacion;
    RETURN @Prom;
END;
GO

/* 3 — Total de inscritos en una actividad */
CREATE FUNCTION Voluntariado.FN_TotalInscritos(@ActividadID INT)
RETURNS INT
AS
BEGIN
    DECLARE @Total INT;
    SELECT @Total = COUNT(*)
    FROM  Voluntariado.Inscripcion
    WHERE ActividadID = @ActividadID;
    RETURN @Total;
END;
GO

/* 4 — Certificados emitidos a un voluntario */
CREATE FUNCTION Certificados.FN_TotalCertificados(@VoluntarioID INT)
RETURNS INT
AS
BEGIN
    DECLARE @Total INT;
    SELECT @Total = COUNT(*)
    FROM  Certificados.Certificado
    WHERE VoluntarioID = @VoluntarioID;
    RETURN @Total;
END;
GO

/* 5 — Conteo de usuarios activos en el sistema */
CREATE FUNCTION Seguridad.FN_UsuariosActivos()
RETURNS INT
AS
BEGIN
    DECLARE @Total INT;
    SELECT @Total = COUNT(*)
    FROM  Seguridad.Usuario
    WHERE Estado = 1;
    RETURN @Total;
END;

Triggers
INSTEAD OF INSERT · AFTER INSERT · AFTER UPDATE
Los Triggers se disparan automáticamente ante eventos DML. Se usan aquí para validar cupos, generar códigos automáticos, registrar auditoría, rechazar datos inválidos y actualizar estados.
TRIGGER INSTEAD OF INSERT TR_ValidarCupo — Control de aforo máximo
▼ 
CREATE TRIGGER TR_ValidarCupo
ON  Voluntariado.Inscripcion
INSTEAD OF INSERT
AS
BEGIN
    -- JOIN entre inserted y Actividad para verificar cupo
    IF EXISTS (
        SELECT 1
        FROM inserted i
        JOIN Voluntariado.Actividad a
          ON i.ActividadID = a.ActividadID
        WHERE (
            SELECT COUNT(*)
            FROM Voluntariado.Inscripcion
            WHERE ActividadID = i.ActividadID
        ) >= a.CupoMaximo
    )
    BEGIN
        RAISERROR('Cupo lleno.', 16, 1);
        RETURN;
    END

    -- Si hay cupo disponible, ejecuta el INSERT real
    INSERT INTO Voluntariado.Inscripcion(VoluntarioID, ActividadID)
    SELECT VoluntarioID, ActividadID FROM inserted;
END;
TRIGGER AFTER INSERT → UPDATE TR_CodigoCertificado — Código automático
▼ 
CREATE TRIGGER TR_CodigoCertificado
ON  Certificados.Certificado
AFTER INSERT
AS
BEGIN
    -- UPDATE automático que genera el código: BIO-{ID}-{AÑO}
    UPDATE c
    SET CodigoCertificado = CONCAT('BIO-', c.CertificadoID, '-', YEAR(GETDATE()))
    FROM Certificados.Certificado c
    JOIN inserted i ON c.CertificadoID = i.CertificadoID;
END;
TRIGGER AFTER INSERT TR_AuditoriaDonacion — Log automático
▼ 
CREATE TRIGGER TR_AuditoriaDonacion
ON  Donaciones.Donacion
AFTER INSERT
AS
BEGIN
    -- Inserta un registro de auditoría por cada donación nueva
    INSERT INTO Seguridad.Auditoria(UsuarioID, Accion, Tabla)
    SELECT NULL, 'INSERT', 'Donacion'
    FROM inserted;
END;

-- Verificar auditoría:
SELECT * FROM Seguridad.Auditoria;
TRIGGER INSTEAD OF INSERT TR_ValidarMontoDonacion — Rechazar montos negativos
▼ 
CREATE TRIGGER TR_ValidarMontoDonacion
ON  Donaciones.Donacion
INSTEAD OF INSERT
AS
BEGIN
    -- Bloquea INSERT si el monto es 0 o negativo
    IF EXISTS (SELECT 1 FROM inserted WHERE Monto <= 0)
    BEGIN
        RAISERROR('Monto inválido.', 16, 1);
        RETURN;
    END

    -- Monto válido → INSERT real
    INSERT INTO Donaciones.Donacion(PersonaID, Monto, MetodoPago)
    SELECT PersonaID, Monto, MetodoPago FROM inserted;
END;

-- Prueba con monto negativo (debe fallar):
INSERT INTO Donaciones.Donacion (PersonaID, Monto, MetodoPago)
VALUES (1, -1000, 'Efectivo');
TRIGGER AFTER UPDATE TR_FinalizarActividad — Cambio de estado automático
▼ 
CREATE TRIGGER TR_FinalizarActividad
ON  Voluntariado.Actividad
AFTER UPDATE
AS
BEGIN
    -- Si la FechaFin ya pasó, marca la actividad como Finalizada
    UPDATE Voluntariado.Actividad
    SET   Estado = 'Finalizada'
    WHERE FechaFin < GETDATE();
END;

-- Verificar estado:
SELECT ActividadID, Estado
FROM  Voluntariado.Actividad
WHERE ActividadID = 1;

Vistas (Views)
CREATE VIEW · SELECT · JOIN · GROUP BY
Las Vistas simplifican consultas complejas y ofrecen una capa de abstracción sobre las tablas base. Cada vista incluye JOINs entre esquemas y puede consultarse con un simple SELECT *.
VIEW SELECT 5 Vistas: VoluntariosActivos · ActividadesConInscritos · DonacionesConfirmadas · CertificadosEmitidos · UsuariosConRol
▼ 
/* 1 — Voluntarios activos con datos personales */
CREATE VIEW Voluntariado.VW_VoluntariosActivos AS
SELECT v.VoluntarioID, p.PrimerNombre, p.PrimerApellido,
       p.Email, v.FechaIngreso
FROM      Personas.Voluntario v
JOIN      Personas.Persona    p ON v.PersonaID = p.PersonaID
WHERE v.Estado = 'Activo';

SELECT * FROM Voluntariado.VW_VoluntariosActivos;

/* 2 — Actividades con conteo de inscritos (LEFT JOIN + GROUP BY) */
CREATE VIEW Voluntariado.VW_ActividadesConInscritos AS
SELECT a.ActividadID, a.NombreActividad, a.CupoMaximo,
       COUNT(i.InscripcionID) AS TotalInscritos
FROM      Voluntariado.Actividad   a
LEFT JOIN Voluntariado.Inscripcion i
    ON a.ActividadID = i.ActividadID
GROUP BY a.ActividadID, a.NombreActividad, a.CupoMaximo;

SELECT * FROM Voluntariado.VW_ActividadesConInscritos;

/* 3 — Donaciones confirmadas filtradas por estado */
CREATE VIEW Donaciones.VW_DonacionesConfirmadas AS
SELECT d.DonacionID, p.PrimerNombre, p.PrimerApellido,
       d.Monto, d.MetodoPago, d.FechaDonacion
FROM  Donaciones.Donacion d
JOIN  Personas.Persona    p ON d.PersonaID = p.PersonaID
WHERE d.Estado = 'Confirmada';

SELECT * FROM Donaciones.VW_DonacionesConfirmadas;

/* 4 — Certificados con nombre de voluntario y actividad (multi-JOIN) */
CREATE VIEW Certificados.VW_CertificadosEmitidos AS
SELECT c.CertificadoID, c.CodigoCertificado,
       p.PrimerNombre, p.PrimerApellido,
       a.NombreActividad, c.FechaEmision
FROM  Certificados.Certificado  c
JOIN  Personas.Voluntario        v ON c.VoluntarioID = v.VoluntarioID
JOIN  Personas.Persona           p ON v.PersonaID    = p.PersonaID
JOIN  Voluntariado.Actividad     a ON c.ActividadID  = a.ActividadID;

SELECT * FROM Certificados.VW_CertificadosEmitidos;

/* 5 — Usuarios del sistema con su rol asignado */
CREATE VIEW Seguridad.VW_UsuariosConRol AS
SELECT u.UsuarioID, u.NombreUsuario,
       r.NombreRol, u.Estado
FROM  Seguridad.Usuario u
JOIN  Seguridad.Rol     r ON u.RolID = r.RolID;

SELECT * FROM Seguridad.VW_UsuariosConRol;

Transacciones
BEGIN TRANSACTION · COMMIT · ROLLBACK · TRY/CATCH
Las Transacciones garantizan atomicidad: si algo falla, el ROLLBACK deshace todos los cambios. El bloque TRY/CATCH captura errores en tiempo de ejecución.
TX INSERT UPDATE 5 Transacciones: Donación · Voluntario · Inscripción+Certificado · Usuario · Estado Actividad
▼ 
/* TX 1 — Registrar donación */
BEGIN TRY
    BEGIN TRANSACTION;

    INSERT INTO Donaciones.Donacion(PersonaID, Monto, MetodoPago)
    VALUES (191, 50000, 'PSE');

    COMMIT;
    PRINT 'TRANSACCION COMPLETADA';
END TRY
BEGIN CATCH
    PRINT ERROR_MESSAGE();
    ROLLBACK;
END CATCH;

/* TX 2 — Registrar voluntario */
BEGIN TRY
    BEGIN TRANSACTION;

    INSERT INTO Personas.Voluntario(PersonaID, FechaIngreso)
    VALUES (190, GETDATE());

    COMMIT;
END TRY
BEGIN CATCH
    PRINT ERROR_MESSAGE();
    ROLLBACK;
END CATCH;

/* TX 3 — Inscripción + Certificado (operación atómica doble) */
BEGIN TRY
    BEGIN TRANSACTION;

    INSERT INTO Voluntariado.Inscripcion(VoluntarioID, ActividadID)
    VALUES (1, 1);

    INSERT INTO Certificados.Certificado(VoluntarioID, ActividadID)
    VALUES (1, 1);

    COMMIT;
END TRY
BEGIN CATCH
    ROLLBACK;
END CATCH;

/* TX 4 — Crear usuario de seguridad */
BEGIN TRY
    BEGIN TRANSACTION;

    INSERT INTO Seguridad.Usuario(PersonaID, NombreUsuario, Contrasena, RolID)
    VALUES (1, 'luisr', '123456', 1);

    COMMIT;
END TRY
BEGIN CATCH
    ROLLBACK;
END CATCH;

/* TX 5 — Actualizar estado de actividad */
BEGIN TRY
    BEGIN TRANSACTION;

    UPDATE Voluntariado.Actividad
    SET   Estado = 'Cancelada'
    WHERE ActividadID = 1;

    COMMIT;
END TRY
BEGIN CATCH
    ROLLBACK;
END CATCH;
